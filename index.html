' checked' : ''); grid.appendChild(div); }); } // 喝水打卡 function checkInToday() { const nextIndex = waterRecord.indexOf(0); if (nextIndex !== -1) { waterRecord[nextIndex] = 1; localStorage.setItem('water_data', JSON.stringify(waterRecord)); renderWaterGrid(); } }
